#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int T;
    cin >> T;

    for (int t = 0; t < T; ++t) {
        int n;
        cin >> n;

        vector<int> sequence(n);
        vector<int> dp(n);

        for (int i = 0; i < n; ++i) {
            cin >> sequence[i];
            dp[i] = sequence[i];
        }




        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                if (sequence[i] > sequence[j]) {
                    dp[i] = max(dp[i], dp[j] + sequence[i]);
                }
            }
        }

        int max_sum = *max_element(dp.begin(), dp.end());


        cout << "case #" << t << ":" << endl;
        cout << max_sum << endl;
    }

    return 0;
}
